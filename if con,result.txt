#include <stdio.h>

int main(){
    int a,b;
    char op;
    scanf("%d %c %d",&a,&op,&b);
    
    switch(op){
        case '+':
        printf("Result = %d ",a + b);
        break;
        
        case '-':
        printf("Result = %d",a - b);
        break;
        
        case '*':
        printf("Result = %d", a * b);
        break;
        
        case '/':
        if(b!=0)
        printf("Result = %d", a/b);
        else
        printf("Division by zero Error");
        break;
        
        default:
        printf("Invalid operator");
    }
    return 0;
    
    }

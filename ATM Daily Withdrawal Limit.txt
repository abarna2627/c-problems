#include <stdio.h>

int main() {
    int N, i;
    int withdrawal;
    int total = 0;

   
    scanf("%d", &N);

    for(i = 0; i < N; i++) {
        scanf("%d", &withdrawal);
        total += withdrawal;
    }

    if(total > 10000)
        printf("Limit Exceeded\n");
    else
        printf("Approved\n");

    return 0;
}

#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int ticket;
    int Total_collection = 0;

    for (int i = 0; i < N; i++) {
        scanf("%d", &ticket);

        if (ticket < 0) {
            ticket = 0;  
        }

        Total_collection += ticket;
    }

    printf("Total collection: $%d\n", Total_collection);
    return 0;
}

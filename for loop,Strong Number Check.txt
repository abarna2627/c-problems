#include <stdio.h>

int main() {
    int N, temp, remainder;
    int sum = 0;
    
    scanf("%d", &N);
    temp = N;

    while (temp > 0) {
        int fact = 1;
        remainder = temp % 10;

        for (int i = 1; i <= remainder; i++) {
            fact = fact * i;
        }

        sum = sum + fact;
        temp = temp / 10;
    }

    if (sum == N) {
        printf("Yes");
    } else {
        printf("No");
    }

    return 0;
}

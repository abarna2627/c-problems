#include <stdio.h>
int main() 
{
    int a,b;
    scanf("%d%d",&a,&b);
    printf("Quotient=%d Remainder=%d\n",a/b,a%b);
    int c,d;
    scanf("%d%d",&c,&d);
    printf("Quotient=%d Remainder=%d\n",c/d,c%d);
    int e,f;
    scanf("%d%d",&e,&f);
    printf("Quotient=%d Remainder=%d\n",e/f,e%f);
    
    return 0;
}
